class MulLayer:
    def __init__(self):
        self.x = None
        self.y = None
        
    def forward(self, x, y):
        self.x = x # self  (backward )
        self.y = y
        out = x * y
        
        return out
    
    def backward(self, dout):
        dx = dout * self.y # x y 
        dy = dout * self.x
        
        return dx, dy

class Relu:
    def __init__(self):
        self.mask = None # mask 0 1   .  x  0 True  False .
        
    def forward(self, x):
        self.mask = (x<=0) #  
        out = x.copy() # x   out 
        out[self.mask] = 0 # x  0 (mask  True)  0 .
        
        return out
    
    def bacward(self, dout):
        dout[self.mask] = 0 # x  0   0 .
        dx = dout #   dout   .
        
        return dx

class Sigmoid:
    def __init__(self):
        self.out = None
        
    def forward(self, x):
        out = 1 / (1+np.exp(-x))
        self.out = out #   self.out     .
        
        return out
    
    def backward(self, dout):
        dx = dout * (1.0 - self.out) * self.out # dL/dy * (1-y) * y
        
        return dx

class Affine:
    def __init__(self, W, b):
        self.W = W
        self.b = b
        self.x = None
        self.dW = None
        self.db = None
        
    def forward(self,x):
        self.x = x
        out = np.dot(x, self.W) +self.b
        
        return out
    
    def backward(self, dout):
        dx = np.dot(dout, self.W.T) # T 
        self.dW = np.dot(self.x.T, dout) # weight 
        self.db = np.sum(dout, axis=0)
        
        return dx

import numpy as np

def softmax(a):
    c = np.max(a)
    exp_a = np.exp(a-c) #  c   .
    sum_exp_a = np.sum(exp_a)
    y = exp_a / sum_exp_a
    
    return y

def corss_entropy_error(y, t):
    delta = 1e-7 #  . 0 .
    if y.ndim == 1: # y  , t  y 1      reshape  .
        t = t.reshape(1, t.size)
        y = y.reshape(1,y.size)
        
    batch_size = y.shape[0] #    y shape     .
    return -np.sum(t*np.log(y + delta)) / batch_size # CEE    batch_size   CEE .

# ch4 TwoLayerNet  .
class TwoLayerNet:
    def __init__(self, input_size, hidden_size, output_size, weight_init_std = 0.01): 
        #   .
        self.params = {}
        self.params['W1'] = weight_init_std * np.random.randn(input_size, hidden_size) # input  hidden_size 
        self.params['b1'] = np.zeros(hidden_size) # hidden_size 
        self.params['W2'] = weight_init_std * np.random.randn(hidden_size, output_size) # hidden_size  output_size 
        self.params['b2'] = np.zeros(output_size) # output_size .
        
        # 
        self.layers = OrderedDict() #    .       forward   .
        #      .
        # Affine  Relu             .
        self.layers['Affine1'] = Affine(self.params['W1'], self.params['b1'])
        self.layers['Relu'] = Relu()
        self.layers['Affine2'] = Affine(self.params['W2'], self.params['b2'])
        
        self.lastLayer = SoftmaxWithLoss()
        
    def predict(self, x):
        for layer in self.layers.values():
            x = layer.forward(x)
        
        return x # self.layers     
    
    # x  t 
    def loss(self, x, t):
        y = self.predict(x)
        
        return self.lastLayer.forward(y,t) # init  lastLayer SoftmaxWithLoss  .
    
    def accuracy(self, x, t):
        y = self.predict(x)
        y = np.argmax(y, axis = 1) # y   argmax .
        if t.ndim != 1: t = np.argmax(t, axis = 1) # ndim 1  t argmax 
        
        accuracy = np.sum(y == t) / float(x.shape[0]) # y t     x     .
        return accuracy
    
    # x  t  .
    #  back propagation .
    def numerical_gradient(self, x, t): #    .
        loss_W = lambda W: self.loss(x,t) # gradient(f,x)  f(x)   W   x, t loss.
        
        grads = {}
        grads['W1'] = numerical_gradient(loss_W, self.params['W1'])
        grads['b1'] = numerical_gradient(loss_W, self.params['b1'])
        grads['W2'] = numerical_gradient(loss_W, self.params['W2'])
        grads['b2'] = numerical_gradient(loss_W, self.params['b2'])
        
        return grads
    
    def gradient(self, x, t):
        # 
        self.loss(x, t)
        
        # 
        dout = 1
        dout = self.lastLayer.backward(dout)
        
        layers = list(self.layers.values())
        layers.reverse()
        for layer in layers:
            dout = layer.backward(dout)
            
        #  
        grads = {}
        grads['W1'] = self.layers['Affine1'].dW
        grads['b1'] = self.layers['Affine1'].db
        grads['W2'] = self.layers['Affine2'].dW
        grads['b2'] = self.layers['Affine2'].db
        
        return grads
